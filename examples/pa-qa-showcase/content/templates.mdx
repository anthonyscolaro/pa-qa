### Example Test Structure

<CodeDemo
  title="React Component Test Example"
  language="typescript"
  runnable={true}
  code={`// tests/unit/components/Button.test.tsx
import { render, screen, fireEvent } from '@testing-library/react'
import { describe, it, expect, vi } from 'vitest'
import { Button } from '@/components/Button'

describe('Button Component', () => {
  it('renders with correct text and variant', () => {
    render(<Button variant="primary">Click me</Button>)
    
    const button = screen.getByRole('button', { name: /click me/i })
    expect(button).toBeInTheDocument()
    expect(button).toHaveClass('btn-primary')
  })

  it('calls onClick handler when clicked', () => {
    const handleClick = vi.fn()
    render(<Button onClick={handleClick}>Click me</Button>)
    
    fireEvent.click(screen.getByRole('button'))
    expect(handleClick).toHaveBeenCalledOnce()
  })

  it('is disabled when loading', () => {
    render(<Button loading>Click me</Button>)
    
    const button = screen.getByRole('button')
    expect(button).toBeDisabled()
    expect(screen.getByText(/loading/i)).toBeInTheDocument()
  })
})`}

@pytest.mark.asyncio
class TestUserService:
    async def test_create_user_success(

        user_service: UserService

        user_data = UserFactory.build()

        created_user = await user_service.create_user(user_data)
        
        # Assert
        assert created_user.id is not None
        assert created_user.email == user_data.email
        assert created_user.is_active is True
        
        # Verify in database
        db_user = await db_session.get(User, created_user.id)
        assert db_user is not None
        assert db_user.email == user_data.email

    async def test_create_user_duplicate_email(
        self,
        db_session: AsyncSession, 
        user_service: UserService
    ):
        # Arrange - create existing user
        existing_user = await UserFactory.create_async(
            session=db_session,

        )
        
        new_user_data = UserFactory.build(email="test@example.com")
        
        # Act & Assert
        with pytest.raises(ValueError, match="Email already exists"):
            await user_service.create_user(new_user_data)

// tests/unit/UserManagerTest.php
use PHPUnit\Framework\TestCase;
use Brain\Monkey\Functions;
use Brain\Monkey\Actions;
use Brain\Monkey\Filters;

class UserManagerTest extends WP_UnitTestCase {
    private $user_manager;
    
    public function setUp(): void {
        parent::setUp();
        $this->user_manager = new UserManager();
    }
    
    public function test_create_user_with_valid_data(): void {
        // Arrange
        $user_data = [
            'user_login' => 'testuser',
            'user_email' => 'test@example.com',
            'user_pass' => 'securepassword123',
            'display_name' => 'Test User'
        ];
        
        // Act
        $user_id = $this->user_manager->create_user($user_data);
        
        // Assert
        $this->assertIsInt($user_id);
        $this->assertGreaterThan(0, $user_id);
        
        $user = get_user_by('id', $user_id);
        $this->assertEquals('test@example.com', $user->user_email);
        $this->assertEquals('Test User', $user->display_name);
    }
    
    public function test_create_user_with_invalid_email(): void {
        // Arrange
        $user_data = [
            'user_login' => 'testuser',
            'user_email' => 'invalid-email',
            'user_pass' => 'securepassword123'
        ];
        
        // Act & Assert
        $this->expectException(InvalidArgumentException::class);
        $this->expectExceptionMessage('Invalid email address');
        
        $this->user_manager->create_user($user_data);
    }
    
    public function test_user_registration_triggers_welcome_email(): void {
        // Arrange
        $user_data = [
            'user_login' => 'newuser',
            'user_email' => 'newuser@example.com',
            'user_pass' => 'password123'
        ];
        
        // Mock email sending
        Functions\expect('wp_mail')
            ->once()
            ->with(
                'newuser@example.com',
                'Welcome to Our Site',
                Mockery::type('string')
            )
            ->andReturn(true);
        
        // Act
        $user_id = $this->user_manager->create_user($user_data);
        
        // Assert - wp_mail expectation is automatically verified
        $this->assertGreaterThan(0, $user_id);
    }
    
    public function tearDown(): void {
        // Clean up test users
        $users = get_users(['meta_key' => 'test_user', 'meta_value' => true]);
        foreach ($users as $user) {
            wp_delete_user($user->ID);
        }
        
        parent::tearDown();
    }
}`}
